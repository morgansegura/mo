## Yes! Some code!

Here is the `Dump` component!

<div className="flex">
    <div className="lg:w-1/3">This is where the info will go</div>
    <div className="lg:w-2/3">This is where the images will go</div>
</div>

```jsx
import React from "react"

const Dump = props => (
    <div
        style={{
            fontSize: 20,
            border: "1px solid #efefef",
            padding: 10,
            background: "white",
        }}
    >
        {Object.entries(props).map(([key, val]) => (
            <pre key={key}>
                <strong style={{ color: "white", background: "red" }}>
                    {key} ?
                </strong>
                {JSON.stringify(val, "", " ")}
            </pre>
        ))}
    </div>
)

export default Dump
```
